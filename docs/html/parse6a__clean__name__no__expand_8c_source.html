15// if the next one is also space it will check if it's the last space

18int get_rid_of_extra_space(char *temp)
19{
20 int i;
21
22 if (is_space(temp[1]) || temp[1] == '\0')
23 {
24 i = 0;
25 while (is_space(temp[i]))
26 i++;
27 if (temp[i] == '\0')

29 }
30 return (1);
31}
