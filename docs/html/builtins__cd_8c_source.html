177int builtins_cd(char **cmd, char **envp, t_token *token)
178{
179 DIR *dir;
180
181 if (cmd[0] && cmd[1])
182 return (write(2, "minishell: cd: too many arguments\n", 34), 1);
183 if (cmd[0] && !str_equals(cmd[0], "-") && !str_equals(cmd[0], "--"))
184 {
185 if (cmd[0][0] == '-')
186 return (write(2, "minishell: cd: invalid option\n", 30), 125);
187 if (cmd[0][0] == '\0')
188 return (0);
189 dir = opendir(cmd[0]);
190 if (!dir)
191 return (write(2, "minishell: No such file or directory\n", 37), 1);
192 closedir(dir);
193 }
194 if ((!cmd[0] || str_equals(cmd[0], "--")) && token->nmb_of_cmd == 1)
195 return (change_dir(NULL, envp, token));
196 else if (token->nmb_of_cmd == 1)
197 return (change_dir(cmd[0], envp, token));
198 return (0);
199}
