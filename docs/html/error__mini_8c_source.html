83void execve_error(char *cmd, char *path, t_token *token, DIR *dir)
84{
85 token->error[0] = 2;
86 token->exit_code[0] = 127;
87 if (cmd)
88 dir = opendir(cmd);
89 if (!dir)
90 {
91 if (cmd && access(cmd, F_OK) == 0 && access(cmd, X_OK) != 0)
92 {
93 mini_error("Permission denied", cmd, NULL, NULL);
94 token->exit_code[0] = 126;
95 }
96 else
97 command_not_found(cmd);
98 }
99 else
100 {
101 closedir(dir);
102 if (!ft_strncmp(cmd, "..", ft_strlen(path)) && ft_strlen(path) > 2)
103 command_not_found(cmd);
104 else
105 {
106 mini_error("Is a directory", cmd, NULL, NULL);
107 token->exit_code[0] = 126;
108 }
109 }
110}
