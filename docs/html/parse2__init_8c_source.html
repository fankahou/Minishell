31static int check_cmd(char **temp, t_token *token, int error_count)
32{
33 int i;
34 int j;
35 int k;
36
37 i = 0;
38 k = 0;
39 while (temp[i])
40 {
41 j = 0;
42 while (is_space(temp[i][j]))
43 j++;
44 if (is_pipe(temp[i][j]))
45 error_count++;
46 else
47 {
48 error_count = 0;
49 k++;
50 }
51 if (error_count == 2)
52 return (syntax_error("|", token), -1);
53 i++;
54 }
