31static int check_temp(char **temp, int error_count, int k)
32{
33 int i;
34 int j;
35
36 i = 0;
37 while (temp[i])
38 {
39 j = 0;
40 while (is_space(temp[i][j]))
41 j++;
42 if (temp[i][j] == '\0')
43 k++;
44 else if (is_delimiter(temp[i][j], temp[i][j + 1]))
45 error_count++;
46 else
47 {
48 error_count--;
49 k++;
50 }
51 if (error_count > 0)
52 return (syntax_error(&temp[i][j], NULL), 0);
53 i++;
54 }
55 if (i > 0 && is_delimiter(temp[i - 1][j], temp[i - 1][j + 1]) > 1)
56 return (syntax_error("unexpected end of file", NULL), 0);
57 return (k);
58}

152{
