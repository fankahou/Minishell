153int builtins_export(char **cmd, t_token *token, int k, int i)
154{
155 if (cmd[0] && cmd[0][0] == '-' && !str_equals(cmd[0], "--"))
156 return (write(2, "minishell: export: invalid option\n", 34), 2);
157 while (cmd[k])
158 {
159 i = 0;
160 if (!ft_isalpha(cmd[k][i]) && cmd[k][i] != '_')
161 return (write(2, "minishell: export: not a valid identifier\n", 42), 1);
162 while (cmd[k][i])
163 {
164 if (!ft_isalpha(cmd[k][i]) && !ft_isalnum(cmd[k][i])
165 && cmd[k][i] != '_' && cmd[k][i] != '=')
166 return (write(2, "minishell: export: not a valid identifier\n", 42), 1);
167 if (cmd[k][i] == '=')
168 break ;
169 i++;
170 }
171 if (cmd[k][i] == '=' && token->nmb_of_cmd == 1)
172 add_envp(cmd[k], token, 0, 0);
173 else if (token->nmb_of_cmd == 1)
174 add_envp_export(cmd[k], token);
175 k++;
176 }
177 if (k == 0)
178 print_array(token->envp_export);
179 return (0);
