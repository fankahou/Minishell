16static int check_builtins(t_cmds *cmds, t_token *token, int *fd)
17{
18 if (builtins_pipe_fd_out(cmds, fd))
19 return (1);
20 if (str_equals(cmds->cmd[0], "echo"))
21 cmds->exit_code = builtins_echo(&cmds->cmd[1]);
22 else if (str_equals(cmds->cmd[0], "cd"))
23 cmds->exit_code = builtins_cd(&cmds->cmd[1], token->envp, token);
24 else if (str_equals(cmds->cmd[0], "pwd"))
25 cmds->exit_code = builtins_pwd(&cmds->cmd[1]);
26 else if (str_equals(cmds->cmd[0], "export"))
27 cmds->exit_code = builtins_export(&cmds->cmd[1], token, 0, 0);
28 else if (str_equals(cmds->cmd[0], "unset"))
29 cmds->exit_code = builtins_unset(&cmds->cmd[1], token->envp, token, 0);
30 else if (str_equals(cmds->cmd[0], "env"))
31 cmds->exit_code = builtins_env(token->envp, &cmds->cmd[1]);
32 else if (str_equals(cmds->cmd[0], "exit"))
33 {
34 builtins_exit(&cmds->cmd[1], token);
35 cmds->exit_code = token->exit_code[0];
36 }
37 else
38 return (0);
39 if (builtins_pipe_fd_in(cmds, fd))
40 return (1);
41 return (1);
42}

78int input(t_cmds *cmds, t_token *token)
79{
80 int fd[2];
81 char *path;
82
83 if (pipe(fd) == -1)
84 return (perror("pipe failed"), 1);
85 if (cmds->fd[0] == -1)
86 return (empty_pipe(fd));
87 if (check_builtins(cmds, token, fd))
88 return (0);
89 if (cmds->cmd[0])
90 path = get_path(cmds->cmd, token->envp);
91 cmds->pid = fork();
92 if (cmds->pid == -1)
93 {
94 close(fd[0]);
95 close(fd[1]);
96 return (perror("fork failed"), 1);
97 }
98 if (cmds->pid == 0)
99 child(fd, cmds, token, path);
100 else
101 parent(fd);
102 if (path)
103 free(path);
104 return (0);
105}
