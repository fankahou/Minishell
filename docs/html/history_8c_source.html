29int history_add(t_history *history, char *command)
30{
31 debug("Adding new element to history...");
32 if (command[0] == '\n' || command[0] == '\0')
33 return (free(command), debug("Will not add empty line to history."), 0);
34 if (history->length > 0 && str_equals(history->list[history->length - 1],
35 command))
36 {
37 return (free(command), debug("Command is already latest entry."), 0);
38 }
39 if (history->length >= HISTORY_MAX_SIZE)
40 {
41 shift_array(history->list, history->length);
42 history->length--;
43 }
44 history->list[history->length] = command;
45 history->length++;
46 add_history(command);
47 return (0);
48}

66int read_history_file(t_history *history, int fd)
67{
68 char *command;
69
70 if (fd < 0)
71 return (0);
72 debug("Reading history from file...");
73 command = get_next_line(fd);
74 while (command)
75 {
76 if (command[0] == '\0')
77 return (free(command), 0);
78 if (command[ft_strlen(command) - 1] == '\n')
79 command[ft_strlen(command) - 1] = '\0';
80 if (command[0] != '\n' && command[0] != '\0')
81 {
82 if (history_add(history, command))
83 warn("Error adding a command from file!");
84 }
85 command = get_next_line(fd);
86 }
87 get_next_line(-1);
88 return (0);
89}
