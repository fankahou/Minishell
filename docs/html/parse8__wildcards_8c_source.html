53static int filter_prefix(char *temp, char *input, int i, int index)
54{
55 char *prefix;
56 char *pin;
57
58 pin = input;
59 prefix = ft_substr(temp, i, index);
60 if (!prefix)
61 return (perror("ft_substr failed"), 1);
62 if (prefix[0] != '\0' && index == (int)ft_strlen(input)
63 && ft_strncmp(&input[index - ft_strlen(prefix)], prefix,
64 ft_strlen(prefix)))
65 return (free(prefix), 1);
66 else if (prefix[0] != '\0' && i == 0 && ft_strncmp(input, prefix,
67 ft_strlen(prefix)))
68 return (free(prefix), 1);
69 else if (prefix[0] != '\0')
70 {
71 pin = ft_strnstr(pin, prefix, ft_strlen(pin));
72 if (!pin)
73 return (free(prefix), 1);
74 pin = pin + ft_strlen(prefix);
75 }
76 free(prefix);
77 return (0);
78}
