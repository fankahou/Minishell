15// token->redir[1] == 6 == Append mode
16// otherwise will create new
17int outfile(t_token *token, int k)
18{
19 int fd;
20
21 if (token->cmds[k]->fd[1] == -1)
22 return (-1);
23 if (access(token->cmds[k]->outfile, F_OK) == 0)
24 {
25 if (access(token->cmds[k]->outfile, W_OK) != 0)
26 return (perror("minishell: Permission denied"),
27 token->exit_code[0] = 13, -1);
28 }
29 if (token->cmds[k]->redir[1] == 6)
30 fd = open(token->cmds[k]->outfile, O_WRONLY | O_CREAT | O_APPEND, 0664);
31 else
32 {
33 if (access(token->cmds[k]->outfile, F_OK) == 0)
34 unlink(token->cmds[k]->outfile);
35 fd = open(token->cmds[k]->outfile, O_WRONLY | O_CREAT, 0664);
36 }
37 if (fd < 0)
38 open_error(token->cmds[k]->outfile, token, token->cmds[k]->fd, k);
39 return (fd);
40}

84int pipex(t_token *token)
85{
86 int i;
87
88 i = 0;
89 while (i < token->nmb_of_cmd - 1 && token->error[0] == 0)
90 {
91 open_fd(token, i);
92 if (replace_fd(token, i))
93 return (perror("dup2 failed"), 1);
94 if (token->cmds[i]->cmd && input(token->cmds[i], token))
95 return (1);
96 i++;
97 }
98 if (token->cmds[i] && token->error[0] == 0)
99 {
100 open_fd(token, i);
101 if (replace_fd(token, i))
102 return (perror("dup2 failed"), 1);
103 if (token->cmds[i]->fd[0] != -1 && token->cmds[i]->cmd
104 && last_input(token->cmds[i], token))
105 return (1);
106 }
